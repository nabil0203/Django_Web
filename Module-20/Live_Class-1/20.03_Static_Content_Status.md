```py
class Task(models.Model):
    STATUS = [
        ('pending', 'Pending'),
        ('completed', 'Completed'),
    ]
    
    CATEGORY = [
        ('work', 'Work'),
        ('personal', 'Personal'),
        ('other', 'Other'),
    ]

    status = models.CharField(max_length=100, choices = STATUS)             # choices = STATUS -> STATUS ki hobe? Pending/Completed
    category = models.CharField(max_length=100, choices=CATEGORY)           # choices = CATEGORY -> CATEGORY ki hobe? Work/Personal/Other
```
